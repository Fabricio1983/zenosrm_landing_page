LÓGICA DE EQUALIZAÇÃO
// Para cada item, encontra menor preço entre fornecedores
const melhorCombinacao = items.map(item => {
  const menorProposta = propostas
    .filter(p => p.itemId === item.id && p.preco > 0)
    .reduce((a, b) => a.preco < b.preco ? a : b);
  return { ...item, fornecedor: menorProposta.fornecedor, preco: menorProposta.preco };
});
// Total da melhor combinação
const totalCombinacao = melhorCombinacao.reduce((sum, item) => sum + item.preco * item.quantidade, 0);
// Melhor pacote = fornecedor que cotou TODOS os itens com menor total
const pacotes = fornecedores.filter(f => {
  return items.every(item => propostas.find(p => p.itemId === item.id && p.fornecedor === f && p.preco > 0));
}).map(f => ({
  fornecedor: f,
  total: propostas.filter(p => p.fornecedor === f).reduce((sum, p) => sum + p.preco * item.quantidade, 0)
}));
const melhorPacote = pacotes.length > 0 ? pacotes.reduce((a, b) => a.total < b.total ? a : b) : null;
// Economia = diferença entre pacote e combinação
const economia = melhorPacote ? melhorPacote.total - totalCombinacao : 0;
