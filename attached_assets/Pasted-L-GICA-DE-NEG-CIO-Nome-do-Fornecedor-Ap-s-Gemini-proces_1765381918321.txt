Rate Limiting (5/dia)
const STORAGE_KEY = "zeno_demo_usage";
const DAILY_LIMIT = 5;
interface UsageData {
  date: string;      // "2025-01-10"
  count: number;     // 0-5
}
function canPerformEqualization(): boolean {
  const today = new Date().toISOString().split('T')[0];
  const data = localStorage.getItem(STORAGE_KEY);
  if (!data) return true;
  
  const usage: UsageData = JSON.parse(data);
  if (usage.date !== today) return true;
  
  return usage.count < DAILY_LIMIT;
}
function incrementUsage(): void {
  const today = new Date().toISOString().split('T')[0];
  const data = localStorage.getItem(STORAGE_KEY);
  
  if (!data) {
    localStorage.setItem(STORAGE_KEY, JSON.stringify({ date: today, count: 1 }));
    return;
  }
  
  const usage: UsageData = JSON.parse(data);
  if (usage.date !== today) {
    localStorage.setItem(STORAGE_KEY, JSON.stringify({ date: today, count: 1 }));
  } else {
    localStorage.setItem(STORAGE_KEY, JSON.stringify({ date: today, count: usage.count + 1 }));
  }
}
function getRemainingToday(): number {
  const today = new Date().toISOString().split('T')[0];
  const data = localStorage.getItem(STORAGE_KEY);
  
  if (!data) return DAILY_LIMIT;
  
  const usage: UsageData = JSON.parse(data);
  if (usage.date !== today) return DAILY_LIMIT;
  
  return Math.max(0, DAILY_LIMIT - usage.count);
}

Cálculo da Equalização
interface Item {
  descricao: string;
  quantidade: number;
  precoUnitario: number;
}
interface FornecedorData {
  nome: string;
  itens: Item[];
  total: number;
}
// Melhor Combinação (multi-source)
function calcularMelhorCombinacao(fornecedores: FornecedorData[]) {
  const todosItens = fornecedores[0].itens.map(item => item.descricao);
  
  return todosItens.map(descricao => {
    let menorPreco = Infinity;
    let melhorFornecedor = "";
    let quantidade = 0;
    
    fornecedores.forEach(f => {
      const item = f.itens.find(i => i.descricao === descricao);
      if (item && item.precoUnitario < menorPreco) {
        menorPreco = item.precoUnitario;
        melhorFornecedor = f.nome;
        quantidade = item.quantidade;
      }
    });
    
    return {
      descricao,
      fornecedor: melhorFornecedor,
      quantidade,
      precoUnitario: menorPreco,
      precoTotal: menorPreco * quantidade
    };
  });
}
// Melhor Pacote (single-source) - fornecedor que cotou TODOS os itens
function calcularMelhorPacote(fornecedores: FornecedorData[]) {
  const todosItens = fornecedores[0].itens.map(item => item.descricao);
  
  // Filtra fornecedores que cotaram todos os itens
  const completos = fornecedores.filter(f => {
    return todosItens.every(desc => 
      f.itens.some(item => item.descricao === desc && item.precoUnitario > 0)
    );
  });
  
  if (completos.length === 0) return null;
  
  // Retorna o de menor total
  return completos.reduce((menor, atual) => 
    atual.total < menor.total ? atual : menor
  );
}
// Economia
function calcularEconomia(melhorPacote: FornecedorData | null, totalCombinacao: number) {
  if (!melhorPacote) return 0;
  return melhorPacote.total - totalCombinacao;
}
// ROI vs Zeno
const MENSALIDADE_ZENO = 497;
function calcularMesesPagos(economia: number) {
  return (economia / MENSALIDADE_ZENO).toFixed(2);
}
